#42Projects
# Clone all 42 projects onto a new machine
function GitClone42Projects() {
    git clone --recurse-submodules git@github.com:LaherMaciel/42Projects.git
    cd 42Projects
    bash setup.sh
}

# Pull latest changes from all submodule remotes and update parent repo pointers
function GitPull42Projects() {
    local current=$(pwd)
    cd /Users/lahermaciel/Documents/Informatica/42Projects
    bash update.sh
    cd "$current"
}

# Commit and push local changes in every submodule, then update parent repo pointers
function GitPush42Projects() {
    local current=$(pwd)
    local root="/Users/lahermaciel/Documents/Informatica/42Projects"

    echo -n "Commit message: "
    read msg
    if [ -z "$msg" ]; then
        echo "Aborted: commit message cannot be empty."
        return 1
    fi

    git submodule foreach --quiet '
        if [ -n "$(git status --porcelain)" ]; then
            git add -A
            git commit -m "'"$msg"'"
            git push
            echo "[OK] Pushed $name"
        else
            echo "[--] No changes in $name"
        fi
    '

    cd "$root"
    if [ -n "$(git status --porcelain)" ]; then
        git add -A
        git commit -m "$msg"
        git push
        echo "[OK] Parent repo updated"
    fi

    cd "$current"
}
